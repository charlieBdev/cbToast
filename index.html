cbToast Demo

new cbToast({
  title: 'Success',
  icon: true,
  message: 'Item Saved!',
  type: 'success', // default, info, error, warning
  position: 'top-right', // top-left, bottom-left, bottom-right, center
  duration: 3000,  // 0 = persistant
  countdown: true,
  maxStack: 5,
  lightMode: false,
  useBS5Theme: false, // overrides lightMode, uses data-bs-theme
  pauseOnHover: true,
  onClose: () => console.log('Closed!')
});
